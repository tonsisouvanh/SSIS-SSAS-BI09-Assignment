non empty [Dim Date].[Date].[Year] on rows
from [DDS Covid19]
where {[Measures].[Fatal],[Measures].[Positive],[Measures].[Resolved]}

	[Measures].[Fatal],
	[Measures].[Positive],
	[Measures].[Resolved]
  } ON COLUMNS,
 non empty crossjoin([Dim Date].[Date].[Year],[Dim PHU].[Phu Name].members) ON ROWS
FROM [DDS Covid19]
